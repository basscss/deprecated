The CSS

.table {
  display: table;
  width: 100%;
}
.table-fixed {
  table-layout: fixed;
}
.table-cell {
  display: table-cell;
  vertical-align: middle;
}
@media (min-width: 640px){
  .mobile-table {
    display: table;
    width: 100%;
  }
  .mobile-table-fixed {
    table-layout: fixed;
  }
  .mobile-table-cell {
    display: table-cell;
    vertical-align: middle;
  }
}

Basic Usage

<div class="table">
  <div class="table-cell"></div>
  <div class="table-cell"></div>
  <div class="table-cell"></div>
</div>

<div class="mobile-table">
  <div class="mobile-table-cell"></div>
  <div class="mobile-table-cell"></div>
  <div class="mobile-table-cell"></div>
</div>
